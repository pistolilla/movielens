DROP TABLE IF EXISTS movie;
CREATE TABLE IF NOT EXISTS movie (movieId INTEGER NOT NULL, title TEXT NOT NULL);
DROP TABLE IF EXISTS genre;
CREATE TABLE IF NOT EXISTS genre (movieId INTEGER NOT NULL, genre TEXT NOT NULL, UNIQUE(movieId, genre) ON CONFLICT IGNORE);
DROP TABLE IF EXISTS link;
CREATE TABLE IF NOT EXISTS link (movieId INTEGER PRIMARY KEY, imdbId INTEGER, tmdbId INTEGER, UNIQUE(movieId, imdbId, tmdbId) ON CONFLICT IGNORE);
